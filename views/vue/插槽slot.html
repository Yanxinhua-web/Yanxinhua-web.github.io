# 为什么要使用插槽

使组件具有更好的拓展性

# 插槽的基本使用

  <div id="app">
        <cpn><button>按钮</button></cpn>
        <cpn><span>哈哈哈哈</span></cpn>
        <cpn><i>呵呵呵呵</i></cpn>
        <cpn>
            <i>a</i>
            <span>b</span>
            <strong>c</strong>
        </cpn>
    </div>
    <template id="cpn">
        <div>
        //多个则会把多个插入
            <h2>我是组件</h2>
            <div>aaaaaaaaaaaa</div>
            <slot></slot>
        </div>
    </template>
    <script src="../js/vue.js"></script>
    <script>
        let app = new Vue({
            el: "#app",
            data: {

            },
            components: {
                cpn: {
                    template: "#cpn"
                }
            }
        })
    </script>

# 具名插槽

  <div id="app">
        <cpn>
            <!-- 这里的center就是组件内的插槽名字 -->
            <span slot="center">标题</span>
        </cpn>
    </div>
    <template id="cpn">
        <div>
            <slot name="left">左边</slot>
            <slot name="center">中间</slot>
            <slot name="right">右边</slot>
        </div>
    </template>

# 作用域插槽

  • 父组件替换插槽的标签，但是内容由子组件来提供
   <div id="app">
        <cpn>
        </cpn>
        <cpn>
            <!-- slot这个名字 是引用的插槽对象 -->
            <template slot-scope="slot">
                <span v-for="item in slot.data">{{item}}-</span>
            </template>
        </cpn>
        <cpn>
            <template slot-scope="slot">
                <span>{{slot.data.join("*")}}</span>
            </template>
        </cpn>
    </div>
    <template id="cpn">
        <div>
            <slot :data="lg">
                <ul>
                    <li v-for="item in lg">{{item}}</li>
                </ul>
            </slot>
        </div>
    </template>
    <script src="../js/vue.js"></script>
    <script>
        let app = new Vue({
            el: "#app",
            data: {

            },
            components: {
                cpn: {
                    template: "#cpn",
                    data() {
                        return {
                            lg: [1, 2, 3, 4, 5, 6, 7, 8, 9]
                        }
                    }

                }
            }
        })
    </script>
