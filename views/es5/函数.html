                # 函数的参数

                 // 定义一个求两个数之和的函数
 // 我们可以在函数的（）中定义一个或多个形参
 // 形参 形式参数  形式主义 空喊 不办事
 // 函数的实参可以是任意的数据类型
function sum() {
console.log(1 + 1)

}

sum()
 // 不对啊 他只是求1+1 但是我求和啊 死的啊
 // 也就是说 我要在调用的时候才决定 谁加谁，定义的时候不用
 // 形参 只声明了 没有赋值
function sum(a, b) {
console.log(a + b)

}

sum() NAN
 // 在调用函数时候我们可以在括号中赋值，也就是制定实参
 // 实参 实际参数  实际主义 办事
 // 实参会赋值给函数中对应的形参
 
sum(1, 2) 3
sum(4, 5) 6

 // 调用函数时候 浏览器解析器不会检查实参的类型
 // 所以有可能会 接收到非法的参数 如果有可能 需要对参数进行 类型的检查
sum(123, 'hello')
sum(true, false)

 // 调用解析器 也不会检查实参的数量
 // 多余的实参不会被赋值
sum(123, 456, 'hello', true, null)

 // 少的实参会变成undefined
 // 如果形参的数量多于实参的数量 则没有值的形参 则会变成undefined
sum(123) NAN  1+undefined

                # 函数的返回值

                • 可以使用return 来设置函数的返回值
                function sum(a, b, c) {
// alert(a + b + c) //不希望函数弹出来
var d = a + b + c
return d
// alert没有执行 
// 在函数return后的语句都不会执行
// 
alert('aaaaa')
}
sum(1, 2, 3)
//既然有返回 就得有个东西来接收返回的值 返回啥接收啥
var result = sum(7, 8, 9)
log(result)

// return 后面不跟值 返回一个undefined
// return 后面可以跟着任意类型的值 null NAN啥都行
// return 可以用来终止函数
function fun() {
return //等同于 return undefined
}
var f = fun()
log(f)

var sayinfo = sayinfo(getInformation)
